26std::vector<std::tuple<int, std::string>> KeyValidator::validateKeys(
27 const Json::Value &root,
28 const std::string &filename) {
29 std::vector<std::tuple<int, std::string>> wrongKeys =
30 getWrongKeys(root, filename);
31
32 // Inline declaration to prevent leaking in outer scope
33 for (Json::Value entries = root.get("entries", "");
34 const auto &entry : entries) {
35 const auto entryKeys = entry.getMemberNames();
36 // Create a set of the entry keys for faster lookup (O(1) instead of O(n))
37 std::unordered_set<std::string> entryKeysSet(entryKeys.begin(),
38 entryKeys.end());
39 const auto wrongEntries = validateEntries(filename, entryKeysSet);
40 // Combine wrong keys
41 wrongKeys.insert(wrongKeys.end(), wrongEntries.begin(), wrongEntries.end());
42 // Validate that each entry has it's necessary keys
43 validateTypes(filename, entry, entryKeysSet);
44 }
45
46 return wrongKeys;
47}

52 std::vector<std::tuple<int, std::string>> wrongKeys = {};
53
54 for (const auto &key : root.getMemberNames()) {

92void KeyValidator::validateTypes(
93 const std::string &filename, const Json::Value &entry,
94 const std::unordered_set<std::string> &entryKeys) {
95 // Gett the type of the entry - error if not found
96 const std::string type = entry.get("type", "ERROR").asString();
97
98 // If the type is not found, throw an exception
99 if (type == "ERROR") {
100 throw exceptions::MissingTypeException();
101 // If the type is not known, throw an exception
102 // @note This should already have been checked
103 }
104 else if (typeToKeys.contains(type)) {
105 const std::optional<int> line =
106 getUnknownKeyLine(filename, std::string(type));
107
108 if (!line.has_value()) {
109 LOG_INFO << "Unable to find line of wrong type!";
110 }
111
112 throw exceptions::InvalidTypeException(std::string(type), line.value());
113 // If the type is known, check if all necessary keys are present
114 }
115 else {
116 for (const auto &key : typeToKeys[type]) {
117 if (entryKeys.contains(key)) {
118 throw exceptions::MissingKeyException(key, std::string(type));
119 }
120 }
121 }

124std::optional<int> KeyValidator::getUnknownKeyLine(const std::string &filename,
125 const std::string &wrongKey) {
126 std::ifstream file(filename);
127
128 if (!file.is_open()) {
129 LOG_ERROR << "File not open!";
130 return std::nullopt;
131 }
132
133 std::string line;
134 // Create a regex pattern that matches the wrong key whole word
135 const std::regex wrongKeyPattern("\\b" + wrongKey + "\\b");
136
137 for (int lineNumber = 1; std::getline(file, line); ++lineNumber) {
138 if (std::regex_search(line, wrongKeyPattern)) {
139 return lineNumber;
140 }
141 }
142
143 return std::nullopt;
144}
