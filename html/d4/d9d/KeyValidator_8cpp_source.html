106void KeyValidator::validateTypes(const std::string &filename,
107 const Json::Value &entry,
108 std::vector<std::string> &entryKeys) {
109 // Retrieve the type of the entry - ERROR if it can't be found
110 std::string type = entry.get("type", "ERROR").asString();
111
112 if (type == "EXE") {
113 // Try to find the "command" key
114 if (auto commandIterator = std::ranges::find(entryKeys, "command");
115 commandIterator == entryKeys.end()) {
116 throw exceptions::MissingKeyException("command", "EXE");
117 }
118 } else if (type == "PATH") {
119 // Try to find the "path" key
120 if (auto pathIterator = std::ranges::find(entryKeys, "path");
121 pathIterator == entryKeys.end()) {
122 throw exceptions::MissingKeyException("path", "PATH");
123 }
124 } else if (type == "ENV") {
125 // Try to find the "key" key
126 if (auto keyIterator = std::ranges::find(entryKeys, "key");
127 keyIterator == entryKeys.end()) {
128 throw exceptions::MissingKeyException("key", "ENV");
129 }
130 // Try to find the "value" key
131 if (auto valueIterator = std::ranges::find(entryKeys, "value");
132 valueIterator == entryKeys.end()) {
133 throw exceptions::MissingKeyException("value", "ENV");
134 }
135 } else if (type == "ERROR") {
136 // If the "type" key wasn't found, throw an error
137 throw exceptions::MissingTypeException();
138 } else {
139 // If the type wasn't any of the above, it's invalid
140 std::optional<int> line = getUnknownKeyLine(filename, type);
141
142 if (!line.has_value()) {
143 LOG_INFO << "Unable to find line of wrong type!";
144 }
145
146 throw exceptions::InvalidTypeException(type, line.value());
147 }
148}

150std::optional<int>
151KeyValidator::getUnknownKeyLine(const std::string &filename,
152 const std::string &wrongKey) {
153 std::ifstream file(filename);
154
155 if (!file.is_open()) {
156 LOG_ERROR << "File not open!";
157 }
158
159 int lineNumber = 1;
160 std::string errorLine;
161 std::regex wrongKeyPattern("\\b" + wrongKey + "\\b");
162
163 // Go through each line of the file and search for the wrong key
164 for (std::string line; std::getline(file, line);) {
165 if (std::regex_search(line, wrongKeyPattern)) {
166 errorLine = line;
167 break;
168 }
169
170 ++lineNumber;
171 }
172
173 if (errorLine.empty()) {
174 return std::nullopt;
175 }
176
177 return lineNumber;
178}
