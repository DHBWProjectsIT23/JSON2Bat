124 exit(1);
125 }

130std::vector<std::string> validateFiles(const std::vector<std::string> &files) {
131 std::vector<std::string> validFiles;
132 // Reserve space, to avaid reallocating with each valid file
133 validFiles.reserve(files.size());
134
135 for (const std::filesystem::path file : files) {
136 // Check that the file exists
137 if (!std::filesystem::is_regular_file(file)) {
138 LOG_ERROR << "The file \"" << file << "\" does not exist!\n";
139
140 if (files.size() > 1 && !utilities::Utils::askToContinue()) {
141 OUTPUT << "Aborting...\n";
142 LOG_INFO << "Application ended by user Input";
143 exit(1);
144 }
145
146 continue;
147 }
148
149 // Check if the file ends in .json
150 if (file.extension() != ".json") {
151 LOG_WARNING << "The file \"" << file << "\" does not end in \".json\"\n";
152 OUTPUT << "If the file is not in JSON Format, continuing may "
153 "result in\nunexpected behaviour!\n";
154
155 if (!utilities::Utils::askToContinue()) {
156 OUTPUT << "Aborting...\n";
157 LOG_INFO << "Application ended by user Input";
158 exit(1);
159 }
160 }
161
162 validFiles.push_back(file);
163 }
164
165 // Shrinks the vector if invalid files were found
166 validFiles.shrink_to_fit();
167 return validFiles;
168}

170void parseFile(const std::string &file, const std::string &outputDirectory) {
171 parsing::JsonHandler jsonHandler(file);
172 const auto fileData = jsonHandler.getFileData();
173 BatchCreator batchCreator(fileData);
174 const std::shared_ptr<std::stringstream> dataStream =
175 batchCreator.getDataStream();
176 // Full filename is output directory + output file
177 const std::string outputFileName =
178 outputDirectory + fileData->getOutputFile();
179 std::ofstream outFile(outputFileName);
180
181 if (!outFile.good()) {
182 throw exceptions::FailedToOpenFileException(outputFileName);
183 }
184
185 outFile << dataStream->str();
186 OUTPUT << "Done with files!\n";
187}
