144std::vector<std::string> validateFiles(const std::vector<std::string> &files) {
145 std::vector<std::string> validFiles;
146 validFiles.reserve(files.size());
147 for (const std::filesystem::path file : files) {
148 if (!std::filesystem::is_regular_file(file)) {
149 LOG_ERROR << "The file \"" << file << "\" does not exist!\n";
150 if (files.size() > 1 &&
151 !utilities::Utils::askToContinue("Do you want to continue with the "
152 "remaining files? (y/n) ")) {
153 OUTPUT << "Aborting...\n";
154 LOG_INFO << "Application ended by user Input";
155 exit(1);
156 }
157 continue;
158 }
159 if (file.extension() != ".json") {
160 LOG_WARNING << "The file \"" << file << "\" does not end in \".json\"\n";
161 OUTPUT << "If the file is not in JSON Format, continuing may "
162 "result in\nunexpected behaviour!\n";
163 if (!utilities::Utils::askToContinue()) {
164 OUTPUT << "Aborting...\n";
165 LOG_INFO << "Application ended by user Input";
166 exit(1);
167 }
168 }
169 validFiles.push_back(file);
170 }
171 return validFiles;
172}

174void parseFile(const std::string &file, const std::string &outputDirectory) {
175 parsing::JsonHandler jsonHandler(file);
176 auto fileData = jsonHandler.getFileData();
177 BatchCreator batchCreator(fileData);
178 std::shared_ptr<std::stringstream> dataStream = batchCreator.getDataStream();
179 std::string outputFileName = outputDirectory + fileData->getOutputFile();
180 std::ofstream outFile(outputFileName);
181 if (!outFile.good()) {
182 throw exceptions::FailedToOpenFileException(outputFileName);
183 }
184 outFile << dataStream->str();
185 OUTPUT << "Done with files!\n";
186}
